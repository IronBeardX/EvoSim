[lexer]

literals:
    + - * / % ( ) ^ @

tokens:
    INTDIV NUMBER


[parser]

expr =>
    expr + term
    expr - term
    term

term =>
    term * factor
    term / factor
    term % factor
    term INTDIV factor
    factor

factor =>
    - factor
    power

power =>
    atom ^ factor
    atom @ factor
    atom

atom =>
    NUMBER
    ( expr )
