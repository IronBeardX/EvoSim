[lexer]

literals:
    + - * / % ( ) ^ @ < >

tokens:
    INTDIV EQ NEQ GE LE NUMBER


[parser]

comparison =>
    expr EQ expr
    expr NEQ expr
    expr GE expr
    expr LE expr
    expr < expr
    expr > expr

expr =>
    expr + term
    expr - term
    term

term =>
    term * factor
    term / factor
    term % factor
    term INTDIV factor
    factor

factor =>
    - factor
    power

power =>
    atom ^ factor
    atom @ factor
    atom

atom =>
    NUMBER
    ( expr )
