[lexer]

literals:
    + - * / % ( ) ^ @ < >

reserved:
    OR AND NOT TRUE FALSE

tokens:
    INTDIV EQ NEQ GE LE NUMBER


[parser]

disjunction =>
    conjunction OR conjunction
    conjunction

conjunction =>
    negation AND negation
    negation

negation =>
    NOT comparison
    comparison

comparison =>
    expr EQ expr
    expr NEQ expr
    expr GE expr
    expr LE expr
    expr < expr
    expr > expr
    expr

expr =>
    expr + term
    expr - term
    term

term =>
    term * factor
    term / factor
    term % factor
    term INTDIV factor
    factor

factor =>
    - factor
    power

power =>
    atom ^ factor
    atom @ factor
    atom

atom =>
    NUMBER
    TRUE
    FALSE
    ( expr )
